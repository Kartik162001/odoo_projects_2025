# Use official Odoo image (17.0 or your version)
FROM odoo:17.0

# Copy your custom addons folder
COPY ./addons /mnt/extra-addons

# Install any Python dependencies (if needed)
COPY requirements.txt /tmp/requirements.txt
RUN pip install --no-cache-dir -r /tmp/requirements.txt || true

# Expose default Odoo port
EXPOSE 8069

# Run Odoo with both standard and custom addons
CMD ["odoo", "--addons-path=/mnt/extra-addons,/usr/lib/python3/dist-packages/odoo/addons"]
